custom props
the code, using CSS custom properties (MDN)

    <style id="ers-linear-equation-default">
    /***********************************************************/
    /* [MANDATORY] Using Custom Properties to calculate y=mx+b */
    /***********************************************************/
    :root {
        /* NOTE: All values in this block hold pixel values (px),
                 but without the px unit attached. */

        /* 
            Initialize 'internal' custom properties. These properties will hold
            the current value of 100vmin/vmax/vw/vh respectively and are being
            kept up to date with by Javascript (Here for demo only as they will 
            be defined by JS either way).
        */
        --js-vmin: 0;
        --js-vmax: 0;
        --js-vw  : 0;
        --js-vh  : 0;

        /*
            Initial values are two points on a straight line
            p1(x1,y1) and p2(x2,y2) => p1(320,14) and p2(1280,20)

            NOTE: Setting these to 0 will work just as well, but giving them a meaningfull
                  initial value makes more sense as it may save writing CSS code later as
                  [--x1] and [--x2] will hardly/not change (in this document anyway).

                  In this case the variables hold the values for [html:font-size]...
        */

        /*
            GeoGebra online examples 

                ERS sizing example 1 (font-size): ERS sizing example 1
                ERS sizing example 2 (padding)  : ERS sizing example 2
        */
        --x1:  320; --y1: 14; /* p1 => 14px at  320px (100vmin = 320px)  */
        --x2: 1280; --y2: 20; /* p2 => 20px at 1280px (100vmin = 1280px) */

        /* Initially assume that [--x] is minimum viewport size dependend */
        --x: var(--js-vmin); /* current 100vmin JS value */

        /* Some initial value. May be 0 too, but now you have a value you can already use */
        --y: calc(0.625vmin + 0.75rem); /* p1(320,14) p2(1280,20) where b=12/16 to get REM */
        /* BEWARE: [--y] now holds some pixel value (unless 'px' is stripped by browser, not sure),
                   but will be reassigned and then hold a regular decimal!
                   (Maybe just assign 0 after all, to prevent use as 'hidden feature'...) */

        /* NOTE: I always use REM instead of PX with all pros and cons considered,
                 therefore all values of [--y] will be divided by 16 converting
                 PX to REM using below [--factorP2R]. */

        /* Division factor PX to REM */
        --factorP2R: 16;
    }
    /*
    Legend:    m   
               x   
               b   
    */
    /* (1) [DEFAULT] Main Equation for use as all-purpose y=mx+b calculation.

        Point-Slope variation: y = y1 + m(x-x1), converted to REM with [--factorP2R]
        Made default as it is the shortest, saving CPU time.

        NOTE: When in doubt, use this one.
    */
    [data-math],
    [data-math="p-slope"] {
        --y: calc((var(--y1) + (var(--y2) - var(--y1)) / (var(--x2) - var(--x1)) *
                  (var(--x) - var(--x1))) / var(--factorP2R))
    }
    /*
        All you need to do now is assign the value of [--y] to an element's...

        - font-size
        - width and/or height
        - padding and/or margin (or -top, -right, -bottom, -left versions)
        - box-shadow, text-shadow (TODO test browser calc() support)
        - border (TODO test browser calc() support)
        - etc.

        ...with a usage selector (see below examples).
    */

    /*********************************************************/
    /* [OPTIONAL][EXTRA] Alternative ways to calculate [--y] */
    /*********************************************************/
    /* (2) Fully substituted version, converted to REM.
           Slower than (1), use it for learning purposes only */
    [data-math="y=mx+b"] {
        --y: calc(
                  (
                    (var(--y2) - var(--y1)) / (var(--x2) - var(--x1)) *
                    var(--x) +
                    (var(--y1) - (var(--y2) - var(--y1)) / (var(--x2) - var(--x1)) * var(--x1))
                  ) / var(--factorP2R))
    }
    /* (3) Split into main variables, Intercept-Slope of y=mx+b.
           Likely slower than (1) and (2) as it uses 3 calc()'s,
           only use it when you need to access [--m] and/or [--b] */
    [data-math="i-slope"] {
        --m: calc((var(--y2) - var(--y1)) / (var(--x2) - var(--x1)));   /* m = (y2 - y1)/(x2 - x1)           */
        --b: calc(var(--y1) - var(--m) * var(--x1));                    /* b = y1 - (y2 - y1)/(x2 - x1) * x1 */
        --y: calc((var(--m) * var(--x) + var(--b)) / var(--factorP2R)); /* y = mx + b, converted to REM      */
    }
    </style>
